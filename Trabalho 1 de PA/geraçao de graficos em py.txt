import pandas as pd
import matplotlib.pyplot as plt

# Carregar os resultados
df = pd.read_csv("resultados.csv")

# Listar tamanhos e casos
tamanhos = df["Tamanho"].unique()
casos = df["Caso"].unique()

# Gerar um gráfico por caso (Aleatório, Crescente, Decrescente)
for caso in casos:
    df_caso = df[df["Caso"] == caso]
    
    plt.figure(figsize=(10,6))
    for alg in df_caso["Algoritmo"].unique():
        df_alg = df_caso[df_caso["Algoritmo"] == alg]
        plt.plot(df_alg["Tamanho"], df_alg["Tempo_medio_microsegundos"], marker="o", label=alg)
    
    plt.title(f"Desempenho dos algoritmos - Caso {caso}")
    plt.xlabel("Tamanho do vetor")
    plt.ylabel("Tempo médio (µs)")
    plt.legend()
    plt.grid(True)
    plt.savefig(f"grafico_{caso}.png")  # salva gráfico
    plt.show()

print("Gráficos salvos como grafico_Aleatório.png, grafico_Crescente.png, grafico_Decrescente.png")
